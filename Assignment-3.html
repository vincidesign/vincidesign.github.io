Program-3

3.Write a program to take an integer "n" as input from the user and print its factorial. 


#include  //header file for standard input output operations
                   //main program starts

int main()
{
    int n,i;        //n for the number, i for the index in for loop
    int fact=1;     // by default factorial values are multipled, hence value is defined as 1
    printf("\t***PROGRAM TO FIND FACTORIAL***\n");
    printf("Enter the number you want to find out factrorial for: ");
    scanf("%d",&n);

     //for loop begins

    for(i=1;i<=n;i++) {

        fact*=i;
    }

    printf("The factorial of %d is %d",n,fact);
    return 0;

}
